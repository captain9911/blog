### html最基本的结构

```
<!DOCTYPE html>    <!-- 声明文档类型，其中html不区分大小写 --> 
<html>    <!-- 描述文档类型 -->
<head>    <!-- 定义文档的页眉 -->
    <meta charset="utf-8">    <!-- 声明编码方式，防止中文显示为乱码 -->
    <title>这是页面标题</title>    <!-- 作为页面标题，显示在浏览器的标签页上 -->
</head>
<body> <!— 页面内容 -->

</body> 
</html>
```
